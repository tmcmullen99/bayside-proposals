async function loadProposalRequests(clientId) { const el = document.getElementById('v2-prop-requests'); if (!el) return; const [{ data: redesigns }, { data: subs }] = await Promise.all([ supabase.from('proposal_redesign_requests') .select('id, proposal_id, status, homeowner_note, designer_response, created_at, reviewed_at, photo_url') .eq('client_id', clientId) .order('created_at', { ascending: false }), supabase.from('proposal_substitutions') .select('id, proposal_id, status, homeowner_note, designer_response, created_at, reviewed_at') .eq('client_id', clientId) .order('created_at', { ascending: false }), ]); const items = [ ...(redesigns || []).map(r => ({ ...r, kind: 'redesign' })), ...(subs || []).map(s => ({ ...s, kind: 'substitution' })), ].sort((a, b) => new Date(b.created_at) - new Date(a.created_at)); if (!items.length) { el.innerHTML = `
Your requests
📝

No requests yet

Use Request redesign or Change materials on a proposal above to submit a change.

`; return; } el.innerHTML = `
Your requests
${items.map(it => { const status = it.status || 'pending'; const badge = ({ pending: { bg: 'var(--bp-cream)', fg: 'var(--bp-muted)', label: 'Pending' }, reviewing: { bg: '#fef3c7', fg: '#92400e', label: 'In review' }, approved: { bg: '#dcfce7', fg: '#166534', label: 'Approved' }, completed: { bg: 'var(--bp-green)', fg: '#fff', label: 'Completed' }, declined: { bg: '#fee2e2', fg: '#991b1b', label: 'Declined' }, })[status] || { bg: 'var(--bp-cream)', fg: 'var(--bp-muted)', label: status }; const icon = it.kind === 'redesign' ? '✏️' : '🔄'; const kindLabel = it.kind === 'redesign' ? 'Redesign request' : 'Material swap'; return `
${icon} ${kindLabel}
${escapeHtml(it.homeowner_note || '(no note)')}
${it.designer_response ? `
Designer reply: ${escapeHtml(it.designer_response)}
` : ''}
Submitted ${formatDate(it.created_at)}${it.reviewed_at ? ' · reviewed ' + formatDate(it.reviewed_at) : ''}
${badge.label}
`; }).join('')}
`; }
